#include <stdio.h>
#include <windows.h>

int countSequences(int n) {
    int dp[n+1]; // Масив для збереження кількості послідовностей
    dp[0] = 1; // Початкове значення для послідовностей довжиною 0 (нульова послідовність)
    dp[1] = 2; // Початкове значення для послідовностей довжиною 1 (послідовності з однією цифрою)
    dp[2] = 4; // Початкове значення для послідовностей довжиною 2 (послідовності з двома цифрами)

    for (int i = 3; i <= n; i++) {
        // Використовуємо формулу для обчислення кількості послідовностей довжиною i
        dp[i] = (dp[i-1] + dp[i-2] + dp[i-3]) % 12345;
    }

    return dp[n]; // Повертаємо кількість шуканих послідовностей довжиною n
}

int main() {
    SetConsoleCP(1251);
  SetConsoleOutputCP(1251);
    int n;
    printf("Введіть довжину послідовності n: ");
    scanf("%d", &n);

    int result = countSequences(n); // Обчислюємо кількість послідовностей
    printf("Кількість шуканих послідовностей: %d\n", result);

    return 0;
}
